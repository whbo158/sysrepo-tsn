
1. connect to star server

	ssh ${USERNAME}@10.192.208.10

2. reserve board in boardfarm

	bft query -b ls1028ardb
	bft reserve ls1028ardb-4

3. map port with testcenter

	bft port -l ls1028ardb-4
	bft map -m 'LS1028ARDB-4_QSGMII1,spirent1g' -c 

	result:
	$ bft map -m 'LS1028ARDB-4_QSGMII1,spirent1g' -c

	Note: QSGMII1--swp0, QSGMII2--swp1

4. connect board in boardfarm

	bft connect ls1028ardb-4
	
5. reset board and update sdcard

	=> usb start
	=> setenv ethact asix_eth
	=> ping 192.168.2.1
	
	=> tftpboot 82000000 ${USERNAME}/sdcard.img  // tftpboot 82000000 nxf24402/sdcard.img
	=> mmc erase 0 0x200000  // (file size in bytes / 512)
	=> mmc write 82000000 0 0x200000

	=> qixis_reset sd
